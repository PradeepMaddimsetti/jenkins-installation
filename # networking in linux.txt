# enabling and disabling ip
    ip link set eth0 up

    ip link set eth0 mtu 9000
# promisc
    ip link set eth0 promisc on

    ip route add default via <local gateway> dev eth0
    ip route add <ip> via <local gateway>
    ip route add <ip> dev eth0
    ip route delete <ip> via <local gateway>
    ip route get <ip>
